CSS Code 
.slide-left {
  position: relative;
}
.slide-left:before {
  content: "";
  position: absolute;
  z-index: -1;
  top: 0;
  left: 0;
  right: 0;
  bottom: 0;
  background: #dadce0;
  transform: scaleX(0);
  transform-origin: 100% 50%;
  transition: transform 0.3s ease-out;
}
.slide-left:hover:before, .slide-left:focus:before, .slide-left:active:before {
  transform: scaleX(1);
}
